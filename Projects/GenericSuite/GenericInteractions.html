Generic Interactions

Simple and effective interaction system for games

What is it?

Generic Interactions is designed to keep efficient but effective interactions for players in games. It works by using a sphere component to detect overlaps with interactable actors and a dot product to see if we are within interaction range with a Line Of Sight check to ensure there are no obstacles in the way. This works both in first and third person games while also allowing you to enable or disable checks using the Z axis for height.

As an added bonus, the system also include a simple way to be able to carry items by attaching them to sockets on the character through blueprints or code using the included carry component. Both systems are designed to be usable separately but carry does not have an interaction detection system on its own so it's implied that they will be used together unless otherwise not wanted by the end user.

Generic Interactions is both highly customizable and easy to use. It comes with an interface as well that allows for:

  • checks on if the object is permitting interactions
  • feedback before and after the interaction
  • Networked choice on how we want the event to be handled (TBD, will most likely be deprecated for possible user error reasons)
  • Server-side interaction checks

    • How to use it

    To use the component, the user simply needs to add the Observer Component to their character or pawn class, adjust the settings for the range and dot product and bind the interaction input to call the TryInteract function on it. From there the item that we wish to interact with simply needs to implement the Interaction Interface and define the desired behaviour on interaction through blueprints or C++.

    There is also the possibility to limit checks to specific channels through a property on the component as well as tweak the interaction range, line of sight checks and Z axis checks. These are not required but can help customize the experience to the user's needs.

    Examples

    The plugin consists of 3 components, 2 of which are mashed into 1. For that reason I'll be sectioning the examples into 3.

    Observer Component

    Code snipped from Observable component for checking for the most relevant object to the player:

    AActor* UGObserverComponent::GetMostRelevantObservable()
{
    if (Observables.IsEmpty())
    {
        return nullptr;
    }

    /*Location data. Not in the loop since it's not relevant to specific interactables, to avoid doing unnecessary calcs*/
    const AActor* Owner = GetOwner();
    FVector ViewOrigin = Owner->GetActorLocation();
    FVector ViewForward = Owner->GetActorForwardVector();
    
    if (bUseCameraForRotation)
    {
        if (APlayerController* PC = Cast(Owner)->GetLocalViewingPlayerController())
        {
            if (APlayerCameraManager* CameraManager = PC->PlayerCameraManager)
            {
                ViewOrigin = CameraManager->GetCameraLocation();
                ViewForward = CameraManager->GetCameraRotation().Vector();
            }
        }
    }
    
    /*Results for finding the best suited interactable*/
    TWeakObjectPtr ChosenObservable = nullptr;
    float BestScore = -FLT_MAX;
    
    for (const TWeakObjectPtr& Observable : Observables)
    {
        /*Data for current interactable*/
        TObjectPtr CurrentObservable = Observable.Get();
        
        // Don't observe pawns (players/NPCs), and don't observe self
        if (CurrentObservable->IsA()) continue;
        if (CurrentObservable == GetOwner()) continue;

        /*If none of the checks for a valid interactable pass, discard this interactable and move on to the next*/
        float DotProduct = 0;
        if (!DoRelevancyChecks(CurrentObservable, ViewOrigin, ViewForward, DotProduct))
        {
            continue;
        }

        /*Tally up this interactable's score in comparison to the other interactables before and see if it's more relevant*/
        const float Score = CalculateRelevancyScore(CurrentObservable, DotProduct);
        if (Score > BestScore)
        {
            BestScore = Score;
            ChosenObservable = CurrentObservable;
        }
    }
    
    return ChosenObservable.Get();
}

    Carry Component

    Attach to actor snippet:

    void UGCarryComponent::AttachToActor(AActor* ObjectToAttach, FName SocketName)
{
	if (!ObjectToAttach) return;

	if (!GetOwner()->HasAuthority())
	{
		Server_AttachToActor(ObjectToAttach, SocketName);
		return;
	}
	
	APawn* OwnerActor = Cast(GetOwner());
	if (!OwnerActor) return;

	if (bOnlyCarry)
	{
		if (!ObjectToAttach->GetClass()->ImplementsInterface(UCarryInterface::StaticClass())) return;

		if(!ICarryInterface::Execute_CanBePickedUp(ObjectToAttach, OwnerActor)) return;
	}
	
	if (USkeletalMeshComponent* SkeletalMeshComp = OwnerActor->FindComponentByClass())
	{
		ObjectToAttach->DisableComponentsSimulatePhysics();
		if (ObjectToAttach->AttachToComponent(SkeletalMeshComp, FAttachmentTransformRules::SnapToTargetNotIncludingScale, SocketName))
		{
			SetAttachedParams(ObjectToAttach, OwnerActor);
			OnItemPickedUp.Broadcast(ObjectToAttach, OwnerActor);
			AttachedObjects.Add(FAttachedItem(SocketName, ObjectToAttach));
			if (NeedsInterfaceCheck(ObjectToAttach))
			{
				ICarryInterface::Execute_OnPickedUp(ObjectToAttach, OwnerActor);
			}
			return;
		}
	}

	if (UStaticMeshComponent* StaticMeshComp = OwnerActor->FindComponentByClass())
	{
		ObjectToAttach->DisableComponentsSimulatePhysics();
		if (ObjectToAttach->AttachToComponent(StaticMeshComp,  FAttachmentTransformRules::SnapToTargetNotIncludingScale, SocketName))
		{
			SetAttachedParams(ObjectToAttach, OwnerActor);
			OnItemPickedUp.Broadcast(ObjectToAttach, OwnerActor);
			AttachedObjects.Add(FAttachedItem(SocketName, ObjectToAttach));
			if (NeedsInterfaceCheck(ObjectToAttach))
			{
				ICarryInterface::Execute_OnPickedUp(ObjectToAttach, OwnerActor);
			}
		}
	}
}

    Interactor Component

    Interact code snippet:

    void UGInteractorComponent::Interact_Implementation(AActor* Target, FInteractionOptionHandle InteractionHandle)
{
    if (!GetOwner()->HasAuthority())
    {
        Interact_Server(Target, InteractionHandle);
        return;
    }
    
    if (!IsValid(Target))
    {
        InteractFailed();
        return;
    }
    
    OnInteractTriggered.Broadcast(Target, Cast(GetOwner()));

    APawn* InstigatorPawn = Cast(GetOwner());
    if (!InstigatorPawn) return;
   
    /*Even on the server this should be the pawn of the player that interacted.
     *Pawns all exist on every version and the owner of this component should be the relevant player*/
    if (Target->GetClass()->ImplementsInterface(UInteractorInterface::StaticClass()))
    {
        if (IInteractorInterface::Execute_CanInteract(Target, InstigatorPawn))
        {
            IInteractorInterface::Execute_Interact(Target, InstigatorPawn,  InteractionHandle);
            UE_LOG(LogGInteractor, Log, TEXT("%s: Interactor: %s interacted with %s"),InstigatorPawn->HasAuthority() ? TEXT("Server") : TEXT("Client"), *InstigatorPawn->GetName(), *Target->GetName());
        }
    }
}
